FROM python:3.12

RUN mkdir /Backend
WORKDIR /Backend

ENV PYTHONDONTWRITEBYTECODE 1
ENV PYTHONUNBUFFERED 1  


RUN pip install fastapi uvicorn
RUN pip install --upgrade pip  

COPY . /app
RUN pip install -r requirements.txt
RUN ls -l


EXPOSE 2024

CMD uvicorn index:app --host 127.0.0.1 --port 2024
